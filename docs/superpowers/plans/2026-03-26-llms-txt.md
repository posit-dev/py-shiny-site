```python
"""Generate llms.txt and llms-full.txt from site source files."""

import re


def clean_qmd_content(content: str) -> str:
    """Strip Quarto-specific markup from .qmd content, keeping prose and code."""
    lines = content.split("\n")
    result_lines: list[str] = []
    in_frontmatter = False
    in_raw_html_block = False
    frontmatter_count = 0

    for line in lines:
        stripped = line.strip()

        # Handle YAML frontmatter (first --- ... --- block)
        if stripped == "---" and frontmatter_count < 2:
            frontmatter_count += 1
            in_frontmatter = not in_frontmatter
            continue
        if in_frontmatter:
            continue

        # Handle raw HTML blocks: ```{=html} ... ```
        if stripped.startswith("```{=html}"):
            in_raw_html_block = True
            continue
        if in_raw_html_block:
            if stripped == "```":
                in_raw_html_block = False
            continue

        # Skip Quarto div fences: ::: {.class}, ::::, ::::{.class}, bare :::
        if re.match(r"^:{3,}\s*(\{.*\})?\s*$", stripped):
            continue

        # Convert Quarto/shinylive code block syntax to standard markdown
        m = re.match(r"^```\{(shinylive-)?(\w+)\}", stripped)
        if m:
            lang = m.group(2)
            result_lines.append(f"```{lang}")
            continue

        # Remove shinylive metadata comments (#| key: value)
        if re.match(r"^#\|\s+\w+", stripped):
            continue

        # Remove inline HTML tags
        line = re.sub(r"<[^>]+>", "", line)

        # Skip lines that became empty after HTML removal (were pure HTML)
        if not line.strip() and stripped:
            continue

        result_lines.append(line)

    # Collapse 3+ consecutive blank lines to 2
    text = "\n".join(result_lines)
    text = re.sub(r"\n{3,}", "\n\n", text)
    return text.strip() + "\n"
```

import yaml


def clean_section_name(raw: str) -> str:
    """Strip Quarto markup from sidebar section names.

    Handles: image syntax, HTML spans, __bold__ markers, {.class} attributes.
    """
    text = raw
    # Remove ![alt](url) and optional {.class} attributes
    text = re.sub(r"!\[[^\]]*\]\([^)]*\)(\{[^}]*\})?", "", text)
    # Remove HTML tags
    text = re.sub(r"<[^>]+>", "", text)
    # Extract text from __bold__ markers
    text = re.sub(r"__([^_]+)__", r"\1", text)
    # Remove {.class} attributes that might remain
    text = re.sub(r"\{[^}]*\}", "", text)
    return text.strip()


def extract_title(content: str) -> str | None:
    """Extract title or pagetitle from .qmd YAML frontmatter."""
    match = re.match(r"^---\n(.*?\n)---", content, re.DOTALL)
    if not match:
        return None
    try:
        fm = yaml.safe_load(match.group(1))
    except yaml.YAMLError:
        return None
    if not isinstance(fm, dict):
        return None
    # pagetitle takes precedence over title
    return fm.get("pagetitle") or fm.get("title")
```

from dataclasses import dataclass
from typing import Iterator


@dataclass
class SidebarEntry:
    """A page discovered from a sidebar, with optional section grouping."""
    section: str | None
    file_path: str


def walk_sidebar(
    contents: list, parent_section: str | None = None
) -> Iterator[SidebarEntry]:
    """Recursively walk sidebar contents, yielding SidebarEntry for each page.

    Handles four entry formats:
    - Plain string: "path/to/file.qmd"
    - Section dict: {section: "Name", contents: [...]}
    - Href dict: {text: "...", href: "path"} — skips anchor links (containing #)
    - File dict: {text: "...", file: "path"}
    """
    for entry in contents:
        if isinstance(entry, str):
            yield SidebarEntry(section=parent_section, file_path=entry)

        elif isinstance(entry, dict):
            if "section" in entry:
                section_name = clean_section_name(entry["section"])
                yield from walk_sidebar(
                    entry.get("contents", []), parent_section=section_name
                )

            elif "href" in entry:
                href = entry["href"]
                if "#" not in href:
                    yield SidebarEntry(section=parent_section, file_path=href)

            elif "file" in entry:
                yield SidebarEntry(section=parent_section, file_path=entry["file"])
```

def file_path_to_url(file_path: str, base_url: str) -> str:
    """Convert a .qmd or .html file path to a full URL.

    - index.qmd / index.html → trailing slash (directory URL)
    - other.qmd → other.html
    - Strips leading slash if present
    """
    path = file_path.lstrip("/")
    name = Path(path).name
    if name in ("index.qmd", "index.html"):
        # Directory URL with trailing slash
        parent = str(Path(path).parent)
        if parent == ".":
            return base_url
        return f"{base_url}{parent}/"
    elif path.endswith(".qmd"):
        path = path[:-4] + ".html"
    return f"{base_url}{path}"
```

# Sidebar ID to display name mapping
SIDEBAR_DISPLAY_NAMES = {
    "get-started": "Get Started",

    "layouts": "Layouts",
    "concepts": "Concepts",

# Sidebar IDs to skip (no useful content for llms.txt)

@dataclass
class Page:
    """A single documentation page."""
    title: str
    file_path: str
    content: str  # raw .qmd content


@dataclass
class Subsection:
    """A subsection within a section (e.g., "Inputs" within "Components")."""
    name: str
    pages: list[Page]


@dataclass
class Section:
    """A top-level section (e.g., "Get Started", "Components")."""
    name: str
    pages: list[Page]  # pages not under any subsection
    subsections: list[Subsection]


def build_site_structure(root: Path) -> list[Section]:
    """Parse _quarto.yml and .qmd files to build ordered site structure."""
    quarto_path = root / "_quarto.yml"
    with open(quarto_path) as f:
        config = yaml.safe_load(f)

    sections: list[Section] = []

    # Walk each sidebar
    sidebars = config.get("website", {}).get("sidebar", [])
    for sidebar in sidebars:
        if not isinstance(sidebar, dict):
            continue
        sidebar_id = sidebar.get("id", "")
        if sidebar_id in SKIP_SIDEBAR_IDS:
            continue
        contents = sidebar.get("contents", [])
        if not contents:
            continue

        section_name = SIDEBAR_DISPLAY_NAMES.get(sidebar_id, sidebar_id.replace("-", " ").title())
        entries = list(walk_sidebar(contents))

        # Group entries by section
        top_pages: list[Page] = []
        subsection_map: dict[str, list[Page]] = {}

        for entry in entries:
            page = _load_page(root, entry.file_path)
            if page is None:
                continue
            if entry.section:
                subsection_map.setdefault(entry.section, []).append(page)
            else:
                top_pages.append(page)

        # Preserve subsection order from walk
        seen_subsections: list[str] = []
        for entry in entries:
            if entry.section and entry.section not in seen_subsections:
                seen_subsections.append(entry.section)

        subsections = [
            Subsection(name=name, pages=subsection_map[name])
            for name in seen_subsections
            if name in subsection_map
        ]

        sections.append(Section(
            name=section_name,
            pages=top_pages,
            subsections=subsections,
        ))

    # Discover API sections (generated by quartodoc)
    sections.extend(_build_api_sections(root))

    # Discover templates by directory scan
    templates_section = _build_templates_section(root)
    if templates_section:
        sections.append(templates_section)

    return sections


def _load_page(root: Path, file_path: str) -> Page | None:
    """Read a .qmd file and extract its title and content."""
    clean_path = file_path.lstrip("/")
    # Convert .html refs to .qmd for reading
    if clean_path.endswith(".html"):
        clean_path = clean_path.replace(".html", ".qmd")
    full_path = root / clean_path
    if not full_path.exists():
        return None
    content = full_path.read_text()
    title = extract_title(content)
    if not title:
        # Fallback to filename
        title = full_path.stem.replace("-", " ").title()
    return Page(title=title, file_path=clean_path, content=content)


def _build_templates_section(root: Path) -> Section | None:
    """Discover template pages by scanning templates/*/index.qmd."""
    templates_dir = root / "templates"
    if not templates_dir.exists():
        return None
    pages: list[Page] = []
    for qmd_path in sorted(templates_dir.glob("*/index.qmd")):
        rel_path = str(qmd_path.relative_to(root))
        page = _load_page(root, rel_path)
        if page:
            pages.append(page)
    if not pages:
        return None
    return Section(name="Templates", pages=pages, subsections=[])


def _build_api_sections(root: Path) -> list[Section]:
    """Build API Reference section from generated sidebar files.

    Returns an empty list if quartodoc hasn't been run yet.
    """
    api_subsections: list[Subsection] = []
    for api_name, display_name in [
        ("express", "Shiny Express"),
        ("core", "Shiny Core"),
        ("testing", "Testing"),
    ]:
        sidebar_path = root / "api" / api_name / "_sidebar.yml"
        if not sidebar_path.exists():
            # API index page only (sidebar not generated yet)
            index_page = _load_page(root, f"api/{api_name}/index.qmd")
            if index_page:
                api_subsections.append(
                    Subsection(name=display_name, pages=[index_page])
                )
            continue
        with open(sidebar_path) as f:
            sidebar_config = yaml.safe_load(f)
        if not sidebar_config:
            continue
        # The sidebar file is a list of contents entries
        contents = sidebar_config if isinstance(sidebar_config, list) else sidebar_config.get("contents", [])
        entries = list(walk_sidebar(contents))
        pages: list[Page] = []
        for entry in entries:
            page = _load_page(root, entry.file_path)
            if page:
                pages.append(page)
        if pages:
            api_subsections.append(Subsection(name=display_name, pages=pages))

    if not api_subsections:
        return []
    return [Section(name="API Reference", pages=[], subsections=api_subsections)]
```

```python
BASE_URL = "https://shiny.posit.co/py/"

SITE_DESCRIPTION = (
    "Shiny for Python is a framework for building reactive web applications "
    "and dashboards in pure Python. It features a reactive execution engine, "
    "built-in UI components, and support for generative AI integration."
)


def _write_header() -> str:
    """Generate the common H1 + blockquote header."""
    return f"# Shiny for Python\n\n> {SITE_DESCRIPTION}\n"


def generate_llms_txt(sections: list[Section]) -> str:
    """Generate concise llms.txt with links only."""
    parts = [_write_header()]

    for section in sections:
        parts.append(f"\n## {section.name}\n")

        # Top-level pages (no subsection)
        for page in section.pages:
            url = file_path_to_url(page.file_path, BASE_URL)
            parts.append(f"- [{page.title}]({url})")

        # Subsections
        for subsection in section.subsections:
            parts.append(f"\n### {subsection.name}\n")
            for page in subsection.pages:
                url = file_path_to_url(page.file_path, BASE_URL)
                parts.append(f"- [{page.title}]({url})")

    return "\n".join(parts) + "\n"


def generate_llms_full_txt(sections: list[Section]) -> str:
    """Generate llms-full.txt with full cleaned content."""
    parts = [_write_header()]

    for section in sections:
        parts.append(f"\n## {section.name}\n")

        # Top-level pages
        for i, page in enumerate(section.pages):
            parts.append(f"### {page.title}\n")
            parts.append(clean_qmd_content(page.content))
            if i < len(section.pages) - 1 or section.subsections:
                parts.append("\n---\n")

        # Subsections
        for si, subsection in enumerate(section.subsections):
            parts.append(f"### {subsection.name}\n")
            for pi, page in enumerate(subsection.pages):
                parts.append(f"#### {page.title}\n")
                parts.append(clean_qmd_content(page.content))
                is_last_page = pi == len(subsection.pages) - 1
                is_last_subsection = si == len(section.subsections) - 1
                if not (is_last_page and is_last_subsection):
                    parts.append("\n---\n")

    return "\n".join(parts) + "\n"
```

import sys


def main() -> None:
    """Entry point: generate llms.txt and llms-full.txt in repo root."""
    # Determine repo root (script is in scripts/, so parent is root)
    root = Path(__file__).resolve().parent.parent

    print("Building site structure from _quarto.yml...")
    sections = build_site_structure(root)

    total_pages = sum(
        len(s.pages) + sum(len(sub.pages) for sub in s.subsections)
        for s in sections
    )
    print(f"Found {len(sections)} sections with {total_pages} pages.")

    print("Generating llms.txt...")
    llms_txt = generate_llms_txt(sections)
    (root / "llms.txt").write_text(llms_txt)

    print("Generating llms-full.txt...")
    llms_full_txt = generate_llms_full_txt(sections)
    (root / "llms-full.txt").write_text(llms_full_txt)

    llms_size = len(llms_txt)
    full_size = len(llms_full_txt)
    print(f"Done. llms.txt: {llms_size:,} bytes, llms-full.txt: {full_size:,} bytes")


if __name__ == "__main__":
    main()
